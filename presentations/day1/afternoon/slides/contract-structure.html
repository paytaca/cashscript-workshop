Rules/Constraints

        require(inputPassword == password);
    }

}
Rules mainly determine if you can spend the funds or not

Variables

pragma cashscript ^0.11.0;

contract MyContract (string password) {

    function spend (string inputPassword) {
        int myNumber = 3000;
        string constant myString = 'Bitcoin Cash';

        int i = 0;
        i = i + 1;
        
        require(inputPassword == password);
    }

}
  • CashScript disallows unused variables
  • Final statement of a function should be a require

Control Structures

pragma cashscript ^0.11.0;

contract MyContract (string password) {

    function spend (string inputPassword) {
        int i = 0;
        i = i + 1;

        if (i == 0) {
            require(inputPassword == password + '0');
        } else if (i == 1) {
            require(inputPassword == password + '1');
        } else {
            require(inputPassword == password + 'other');
        }
        
        require(inputPassword == password);
    }

}
